# prompt: create a chart that explores the relationship between car make and city

# Analyze the relationship between Make and City
# For this, we can look at the count of each make within different cities.
# Since there are many cities and makes, let's focus on the top N cities and top N makes.
top_cities = df['City'].value_counts().nlargest(10).index.tolist()
top_makes = df['Make'].value_counts().nlargest(10).index.tolist()

# Filter the dataframe to include only the top cities and makes
df_filtered = df[df['City'].isin(top_cities) & df['Make'].isin(top_makes)]

# Create a cross-tabulation (contingency table)
cross_tab = pd.crosstab(df_filtered['City'], df_filtered['Make'])

# Create a heatmap to visualize the relationship
plt.figure(figsize=(14, 8))
sns.heatmap(cross_tab, annot=True, fmt='d', cmap='YlGnBu')

# Add titles and labels
plt.title('Relationship between Top 10 Car Makes and Top 10 Cities (Count)', fontsize=16)
plt.xlabel('Car Make', fontsize=12)
plt.ylabel('City', fontsize=12)

# Show the plot
plt.tight_layout()
plt.show()
